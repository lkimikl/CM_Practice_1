#Тестовый скрипт для проверки всех команд
#Команда pwd
pwd

#Команда ls без аргументов
ls

#Команда ls с одним аргументом
ls -l

#Команда ls с аргументом в кавычках
ls "-la"

#Команда cd без аргументов
cd

#Команда cd с абсолютным путем
cd /home/user

#Команда cd с относительным путем
cd documents

#Команда cd с путем в кавычках
cd "my folder"

#Снова pwd для проверки
pwd

#Тестирование ошибок - слишком много аргументов
pwd extra
ls -l -a
cd /home /tmp
exit 123

#Корректный exit
exit
